\begin{equation}
  y(x)=0.3*cos(\sqrt{q}*x)
\end{equation}
\begin{equation}
  y(a) = 0 \Rightarrow y(a)=0.3*cos(\sqrt{q}*a) = 0
\end{equation}
\begin{equation}
  \Rightarrow cos(\sqrt{q}*a) = 0
\end{equation}
\begin{equation}
  \Rightarrow \sqrt{q}*a = \pi * n - \frac{\pi}{2}      n \in Z
\end{equation}
We only consider $n=1$
\begin{equation}
  \sqrt{q}*a=\frac{\pi}{2} \Rightarrow a=\frac{\pi}{2*\sqrt{q}}
\end{equation}
